;simple adc interfacing
DATA SEGMENT
 PORTA EQU 00H
 PORTB EQU 02H
 PORTC EQU 04H
 PORT_CON EQU 06H
DATA ENDS
CODE SEGMENT
 MOV AX,DATA
 MOV DS, AX

ORG 0000H
START:

 MOV DX, PORT_CON
 MOV AL, 10010000B
 OUT DX, AL

 MOV AL,00H

   
XX:
 MOV DX,PORTA
 IN AL,DX
 MOV DX, PORTC
 OUT DX,AL
 MOV DX, PORTB
 MOV AL, 00000000B
 OUT DX,AL

 MOV cx, 0ffH
D1: LOOP D1
   MOV DX, PORTB
 MOV AL, 00000001B
 OUT DX,AL
          
           MOV cx, 0ffH
D2: LOOP D2
 JMP XX

CODE ENDS
END
